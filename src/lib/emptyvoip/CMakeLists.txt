CMAKE_MINIMUM_REQUIRED( VERSION 2.8 )
PROJECT( emptyvoip )

INCLUDE( BWStandardProject )

INCLUDE_DIRECTORIES( ${BW_SOURCE_DIR}/client )

ADD_DEFINITIONS( 
	-DVOIP_EXPORTS
)

SET( ALL_SRCS
	voip.cpp
)
SOURCE_GROUP( "" FILES ${ALL_SRCS} )


BW_ADD_LIBRARY( emptyvoip SHARED
	${ALL_SRCS}
)

SET_TARGET_PROPERTIES( emptyvoip PROPERTIES
		DEBUG_OUTPUT_NAME
		"voip_d" )

SET_TARGET_PROPERTIES( emptyvoip PROPERTIES
		HYBRID_OUTPUT_NAME
		"voip_h" )

SET_TARGET_PROPERTIES( emptyvoip PROPERTIES
		CONSUMER_RELEASE_OUTPUT_NAME
		"voip" )

BW_PROJECT_CATEGORY( emptyvoip "BigWorld Libs" )
BW_SET_BINARY_DIR( emptyvoip "${BW_GAME_DIR}/client/${BW_PLATFORM}" )
